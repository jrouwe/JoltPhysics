599// Macro to enable floating point precise mode and to disable fused multiply add instructions
600#if defined(JPH_COMPILER_GCC) || defined(JPH_CROSS_PLATFORM_DETERMINISTIC)
601 // We compile without -ffast-math and -ffp-contract=fast, so we don't need to disable anything
602 #define JPH_PRECISE_MATH_ON
603 #define JPH_PRECISE_MATH_OFF
604#elif defined(JPH_COMPILER_CLANG)
605 // We compile without -ffast-math because pragma float_control(precise, on) doesn't seem to actually negate all of the -ffast-math effects and causes the unit tests to fail (even if the pragma is added to all files)
606 // On clang 14 and later we can turn off float contraction through a pragma (before it was buggy), so if FMA is on we can disable it through this macro
607 #if (defined(JPH_CPU_ARM) && !defined(JPH_PLATFORM_ANDROID) && __clang_major__ >= 16) || (defined(JPH_CPU_X86) && __clang_major__ >= 14)
608 #define JPH_PRECISE_MATH_ON \
609 _Pragma("float_control(precise, on, push)") \
610 _Pragma("clang fp contract(off)")
611 #define JPH_PRECISE_MATH_OFF \
612 _Pragma("float_control(pop)")
613 #elif __clang_major__ >= 14 && (defined(JPH_USE_FMADD) || defined(FP_FAST_FMA))
614 #define JPH_PRECISE_MATH_ON \
615 _Pragma("clang fp contract(off)")
616 #define JPH_PRECISE_MATH_OFF \
617 _Pragma("clang fp contract(on)")
618 #else
619 #define JPH_PRECISE_MATH_ON
620 #define JPH_PRECISE_MATH_OFF
621 #endif
622#elif defined(JPH_COMPILER_MSVC)
623 // Unfortunately there is no way to push the state of fp_contract, so we have to assume it was turned on before JPH_PRECISE_MATH_ON
624 #define JPH_PRECISE_MATH_ON \
625 __pragma(float_control(precise, on, push)) \
626 __pragma(fp_contract(off))
627 #define JPH_PRECISE_MATH_OFF \
628 __pragma(fp_contract(on)) \
629 __pragma(float_control(pop))
