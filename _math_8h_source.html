28inline float CenterAngleAroundZero(float inV)
29{
30 if (inV < -JPH_PI)
31 {
32 do
33 inV += 2.0f * JPH_PI;
34 while (inV < -JPH_PI);
35 }
36 else if (inV > JPH_PI)
37 {
38 do
39 inV -= 2.0f * JPH_PI;
40 while (inV > JPH_PI);
41 }
42 JPH_ASSERT(inV >= -JPH_PI && inV <= JPH_PI);
43 return inV;
44}

164inline uint CountBits(uint32 inValue)
165{
166#if defined(JPH_COMPILER_CLANG) || defined(JPH_COMPILER_GCC)
167 return __builtin_popcount(inValue);
168#elif defined(JPH_COMPILER_MSVC)
169 #if defined(JPH_USE_SSE4_2)
170 return _mm_popcnt_u32(inValue);
171 #elif defined(JPH_USE_NEON) && (_MSC_VER >= 1930) // _CountOneBits not available on MSVC2019
172 return _CountOneBits(inValue);
173 #else
174 inValue = inValue - ((inValue >> 1) & 0x55555555);
175 inValue = (inValue & 0x33333333) + ((inValue >> 2) & 0x33333333);
176 inValue = (inValue + (inValue >> 4)) & 0x0F0F0F0F;
177 return (inValue * 0x01010101) >> 24;
178 #endif
179#else
180 #error Undefined
181#endif
182}
